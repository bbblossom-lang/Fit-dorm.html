        let breathingPaused = false;
        let musicPlaying = false;

            "Success is the sum of small efforts repeated day in and day out.",

        function selectGoal(goal) {
            currentGoal = goal;
            document.getElementById('mainNav').style.display = 'flex';
            showSection('videos');
            
            let message = '';
            switch(goal) {
                case 'lose-weight':
                    message = 'üî• Great choice! Focus on cardio and HIIT workouts for maximum fat burning.';
                    break;
                case 'relax':
                    message = 'üßò‚Äç‚ôÄÔ∏è Perfect! Check out our yoga and stretching videos for stress relief.';
                    break;
                case 'build-strength':
                    message = 'üí™ Awesome! Focus on resistance training and progressive overload.';
                    break;
            }
            document.getElementById('goalMessage').textContent = message;
        }

        function showSection(sectionId) {
            document.querySelectorAll('.section').forEach(section => {
                section.classList.remove('active');
            });
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            document.getElementById(sectionId).classList.add('active');
            
            const clickedBtn = Array.from(document.querySelectorAll('.nav-btn')).find(btn => 
                btn.textContent.toLowerCase().includes(sectionId)
            );
            if (clickedBtn) clickedBtn.classList.add('active');
        }

        // Drag and Drop Functions
        function drag(e) {
            draggedWorkout = e.target.textContent;
            draggedElement = e.target;
            e.target.classList.add('dragging');
        }

        function allowDrop(e) {
            e.preventDefault();
            e.currentTarget.classList.add('drag-over');
        }

        function dragLeave(e) {
            e.currentTarget.classList.remove('drag-over');
        }

        function drop(e) {
            e.preventDefault();
            e.currentTarget.classList.remove('drag-over');
            
            const assignedDiv = document.createElement('div');
            assignedDiv.className = 'assigned-workout';
            assignedDiv.innerHTML = `
                <span>${draggedWorkout}</span>
                <button class="remove-btn" onclick="this.parentElement.remove()">√ó</button>
            `;
            e.currentTarget.appendChild(assignedDiv);
            
            if (draggedElement) {
                draggedElement.classList.remove('dragging');
                draggedElement = null;
            }
        }

        // Touch support for mobile/iPad
        document.addEventListener('DOMContentLoaded', function() {
            const workoutItems = document.querySelectorAll('.workout-item');
            const daySlots = document.querySelectorAll('.day-slot');
            
            workoutItems.forEach(item => {
                // Touch events for iPad
                item.addEventListener('touchstart', function(e) {
                    e.preventDefault();
                    draggedWorkout = this.textContent;
                    this.style.opacity = '0.5';
                });
                
                item.addEventListener('touchend', function(e) {
                    this.style.opacity = '1';
                });
                
                item.addEventListener('touchmove', function(e) {
                    e.preventDefault();
                    const touch = e.touches[0];
                    const element = document.elementFromPoint(touch.clientX, touch.clientY);
                    
                    // Find if we're over a day slot
                    let daySlot = element;
                    while (daySlot && !daySlot.classList.contains('day-slot')) {
                        daySlot = daySlot.parentElement;
                    }
                    
                    // Highlight the day slot
                    daySlots.forEach(slot => slot.classList.remove('drag-over'));
                    if (daySlot && daySlot.classList.contains('day-slot')) {
                        daySlot.classList.add('drag-over');
                    }
                });
            });
            
            daySlots.forEach(slot => {
                slot.addEventListener('touchend', function(e) {
                    if (this.classList.contains('drag-over') && draggedWorkout) {
                        const assignedDiv = document.createElement('div');
                        assignedDiv.className = 'assigned-workout';
                        assignedDiv.innerHTML = `
                            <span>${draggedWorkout}</span>
                            <button class="remove-btn" onclick="this.parentElement.remove()">√ó</button>
                        `;
                        this.appendChild(assignedDiv);
                        this.classList.remove('drag-over');
                        draggedWorkout = '';
                    }
                });
            });
        });

        function savePlan() {
            alert('‚úÖ Your workout plan has been saved! Stay consistent and crush your goals! üí™');
        }

        function newQuote() {
            const randomQuote = quotes[Math.floor(Math.random() * quotes.length)];
            document.getElementById('quoteDisplay').textContent = randomQuote;
        }

        // Breathing Animation Control
        function toggleBreathing() {
            const circle = document.getElementById('breathingCircle');
            const text = document.getElementById('breathingText');
            const btn = document.getElementById('breathingBtn');
            
            breathingPaused = !breathingPaused;
            
            if (breathingPaused) {
                circle.classList.add('paused');
                text.classList.add('paused');
                btn.textContent = '‚ñ∂Ô∏è Resume';
            } else {
                circle.classList.remove('paused');
                text.classList.remove('paused');
                btn.textContent = '‚è∏Ô∏è Pause';
            }
        }

        // Update breathing text
        setInterval(() => {
            if (!breathingPaused) {
                const text = document.getElementById('breathingText');
                const computedStyle = window.getComputedStyle(text);
                const opacity = parseFloat(computedStyle.opacity);
                
                if (opacity > 0.5) {
                    text.textContent = 'Breathe In';
                } else {
                    text.textContent = 'Breathe Out';
                }
            }
        }, 100);

        // Music Control
        function toggleMusic() {
            const music = document.getElementById('backgroundMusic');
            const musicInfo = document.querySelector('.music-info p');
            
            if (musicPlaying) {
                music.pause();
                musicInfo.textContent = 'üéµ Background Music: Click to play';
                musicPlaying = false;
            } else {
                music.play();
                musicInfo.textContent = 'üé∂ Background Music: Calm ambient sounds playing';
                musicPlaying = true;
            }
        }
